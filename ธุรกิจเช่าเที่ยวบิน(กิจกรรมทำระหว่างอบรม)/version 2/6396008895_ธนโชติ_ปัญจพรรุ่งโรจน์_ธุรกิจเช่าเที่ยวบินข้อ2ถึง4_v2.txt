==== กิจกรรม 4: สร้างตารางโดยใช้ SQL ====

CREATE TABLE chapter
(
	CHAR_NUM smallint NOT NULL PRIMARY KEY,
	CHAR_DATE date NOT NULL,
	CHAR_ DESTINATION char(3) NOT NULL,
	CHAR_DISTANCE smallint NOT NULL,
	CHAR_HOURS smallint NOT NULL,
	CHAR_FUEL smallint NOT NULL
);

CREATE TABLE customer
(
	CUS_CODE smallint NOT NULL PRIMARY KEY,
	CUS_FNAME varchar NOT NULL,
	CUS_LNAME varchar NOT NULL,
	CUS_AREACODE smallint NOT NULL,
	CUS_PHONE smallint NOT NULL,
	CUS_BALANCE smallint NOT NULL
);

CREATE TABLE aircraft
(
	AC_NUM smallint NOT NULL PRIMARY KEY,
	AC_TTAF smallint NOT NULL,
	AC_TTEL smallint NOT NULL,
	AC_TTER smallint NOT NULL
);

CREATE TABLE model
(
	MOS_CODE smallint NOT NULL PRIMARY KEY,
	MOD_NAME varchar NOT NULL,
	MOD_SEAT smallint NOT NULL,
	MOD_MANUFACTURER varchar NOT NULL,
	MOD_CHG_MILE smallint NOT NULL,
);

CREATE TABLE pilot
(
	PIL_NUM smallint NOT NULL PRIMARY KEY,
	PIL_LICENSE smallint NOT NULL,
	PIL_RATINGS smallint NOT NULL,
	PIL_MED_TYPE varchar NOT NULL,
	PIL_MED_DATE date NOT NULL,
	PIL_SKILL_DATE date NOT NULL
);

CREATE TABLE employee
(
	EMP_NUM smallint NOT NULL PRIMARY KEY,
	EMP_TITLE smallint NOT NULL,
	EMP_FNAME smallint NOT NULL,
	EMP_LNAME varchar NOT NULL,
	EMP_ADDRESS varchar NOT NULL,
	EMP_DOB date NOT NULL,
	EMP_HIRE_DATE date NOT NULL
);

CREATE TABLE chapter customer
(
	CHAR_NUM smallint NOT NULL REFERENCE chapter (CHAR_NUM),
	CUS_CODE smallint NOT NULL REFERENCE customer (CUS_CODE),
	PRIMARY KEY (CHAR_NUM, CUS_CODE)
);

CREATE TABLE chapter aircraft
(
	CHAR_NUM smallint NOT NULL REFERENCE chapter (CHAR_NUM),
	AC_NUM smallint NOT NULL REFERENCE aircraft (AC_NUM),
	PRIMARY KEY (CHAR_NUM, AC_NUM)
);

CREATE TABLE chapter pilot
(
	CHAR_NUM smallint NOT NULL REFERENCE chapter (CHAR_NUM),
	PIL_NUM smallint NOT NULL REFERENCE pilot (PIL_NUM), 
	COPIL_PIL_NUM smallint REFERENCE pilot (PIL_NUM),
	PRIMARY KEY (CHAR_NUM, PIL_NUM)
);

CREATE TABLE aircraft model
(
	AC_NUM smallint NOT NULL REFERENCE aircraft (AC_NUM),
	MOD_CODE smallint NOT NULL REFERENCE model (MOD_CODE),
	PRIMARY KEY (AC_NUM, MOD_CODE)
);

CREATE TABLE pilot employee
(
	PIL_NUM smallint NOT NULL PRIMARY KEY REFERENCE pilot (PIL_NUM),
	EMP_NUM smallint NOT NULL REFERENCE employee (EMP_NUM),
	PRIMARY KEY (PIL_NUM, EMP_NUM)
);
